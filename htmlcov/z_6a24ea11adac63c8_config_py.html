87def extract_domain(url: str) -> str: 

88 """ 

89 從 URL 中提取網域名稱 

90  

91 Args: 

92 url: 要解析的 URL 

93  

94 Returns: 

95 str: 網域名稱，如果解析失敗返回空字串 

96 """ 

97 try: 

98 # 確保 URL 有協議 

99 if not url.startswith(("http://", "https://")): 

100 url = "http://" + url 

101 

102 parsed = up.urlparse(url) 

103 domain = parsed.netloc.lower() 

104 

105 # 移除端口號 

106 if ":" in domain: 

107 domain = domain.split(":")[0] 

108 

109 # 移除 www. 前綴 

110 if domain.startswith("www."): 

111 domain = domain[4:] 

112 

113 return domain 

114 except Exception: 

115 return ""

117def normalize_domain(domain: str) -> str: 

118 """ 

119 標準化網域名稱 

120  

121 Args: 

122 domain: 原始網域名稱 

123  

124 Returns: 

125 str: 標準化後的網域名稱 

126 """ 

127 try: 

128 domain = domain.lower().strip() 

129 

130 # 移除協議 

131 if domain.startswith(("http://", "https://")): 

132 domain = up.urlparse(domain).netloc 

133 

134 # 移除端口號 

135 if ":" in domain: 

136 domain = domain.split(":")[0] 

137 

138 # 移除 www. 前綴 

139 if domain.startswith("www."): 

140 domain = domain[4:] 

141 

142 # 移除路徑 

143 if "/" in domain: 

144 domain = domain.split("/")[0] 

145 

146 return domain 

147 except Exception: 

148 return ""

150def is_whitelisted(domain: str, whitelist: Set[str]) -> bool: 

151 """ 

152 檢查網域是否在白名單中 

153  

154 Args: 

155 domain: 要檢查的網域 

156 whitelist: 白名單集合 

157  

158 Returns: 

159 bool: 是否在白名單中 

160 """ 

161 try: 

162 domain = normalize_domain(domain) 

163 if not domain: 

164 return False 

165 

166 # 直接匹配 

167 if domain in whitelist: 

168 return True 

169 

170 # 檢查子網域 

171 for whitelisted in whitelist: 

172 if domain.endswith("." + whitelisted): 

173 return True 

174 

175 return False 

176 except Exception: 

177 return False

179def parse_domain_list(domain_str: str) -> Set[str]: 

180 """ 

181 解析網域列表字串 

182  

183 Args: 

184 domain_str: 以逗號分隔的網域列表 

185  

186 Returns: 

187 Set[str]: 標準化後的網域集合 

188 """ 

189 try: 

190 if not domain_str: 

191 return set() 

192 

193 domains = set() 

194 for domain in domain_str.split(","): 

195 normalized = normalize_domain(domain.strip()) 

196 if normalized: 

197 domains.add(normalized) 

198 

199 return domains 

200 except Exception: 

201 return set()

203def validate_domain(domain: str) -> bool: 

204 """ 

205 驗證網域格式是否正確 

206  

207 Args: 

208 domain: 要驗證的網域 

209  

210 Returns: 

211 bool: 是否為有效網域 

212 """ 

213 try: 

214 domain = normalize_domain(domain) 

215 if not domain: 

216 return False 

217 

218 # 基本格式檢查 

219 if not re.match(r'^[a-zA-Z0-9]([a-zA-Z0-9\-]{0,61}[a-zA-Z0-9])?(\.[a-zA-Z0-9]([a-zA-Z0-9\-]{0,61}[a-zA-Z0-9])?)*$', domain): 

220 return False 

221 

222 # 檢查是否包含有效的 TLD 

223 if "." not in domain: 

224 return False 

225 

226 # 檢查 TLD 長度 

227 tld = domain.split(".")[-1] 

228 if len(tld) < 2: 

229 return False 

230 

231 return True 

232 except Exception: 

233 return False

267def format_domain_list(domains: Set[str], max_length: int = 1000) -> str: 

268 """ 

269 格式化網域列表為字串 

270  

271 Args: 

272 domains: 網域集合 

273 max_length: 最大字串長度 

274  

275 Returns: 

276 str: 格式化後的字串 

277 """ 

278 try: 

279 if not domains: 

280 return "（無）" 

281 

282 sorted_domains = sorted(domains) 

283 result = [] 

284 current_length = 0 

285 

286 for domain in sorted_domains: 

287 line = f"• {domain}" 

288 if current_length + len(line) + 1 > max_length: 

289 remaining = len(sorted_domains) - len(result) 

290 if remaining > 0: 

291 result.append(f"... 還有 {remaining} 個網域") 

292 break 

293 

294 result.append(line) 

295 current_length += len(line) + 1 

296 

297 return "\n".join(result) 

298 except Exception: 

299 return "（解析錯誤）"
