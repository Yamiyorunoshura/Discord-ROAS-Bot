29 def __init__(self, bot: discord.Client, db: ActivityDatabase): 

30 """ 

31 初始化背景任務處理器

37 self.bot = bot

39 self.auto_report_task = None

41 def start(self): 

42 """啟動所有背景任務""" 

43 self.auto_report_task = self.auto_report.start() 

44 

45 def stop(self): 

46 """停止所有背景任務""" 

47 if self.auto_report_task and self.auto_report.is_running(): 

48 self.auto_report.cancel()

50 @tasks.loop(minutes=1) 

51 async def auto_report(self): 

52 """ 

53 自動播報排行榜任務 

54 每日在指定時間自動發送排行榜到設定的頻道 

55 """ 

56 try: 

57 # 檢查是否為指定播報時間 

58 now = datetime.now(config.TW_TZ) 

59 if now.hour != config.ACT_REPORT_HOUR or now.minute != 0: 

60 return 

61 

62 # 獲取日期字串 

63 ymd = now.strftime(config.DAY_FMT) 

64 ym = now.strftime(config.MONTH_FMT) 

65 days = int(now.strftime("%d")) 

66 

67 # 獲取所有設定了播報頻道的伺服器 

68 report_channels = await self.db.get_report_channels() 

69 

70 for guild_id, channel_id in report_channels: 

71 await self._process_guild_report(guild_id, channel_id, ymd, ym, days) 

72 

73 except Exception as e: 

74 logger.error(f"【活躍度】自動播報任務執行失敗: {e}") 

75 

76 @auto_report.before_loop 

77 async def _wait_ready(self): 

78 """等待機器人就緒""" 

79 await self.bot.wait_until_ready()

81 async def _process_guild_report(self, guild_id: int, channel_id: int, ymd: str, ym: str, days: int): 

82 """ 

83 處理單一伺服器的排行榜播報 

84  

85 Args: 

86 guild_id: 伺服器 ID 

87 channel_id: 頻道 ID 

88 ymd: 日期字串 (YYYYMMDD) 

89 ym: 月份字串 (YYYYMM) 

90 days: 當月天數 

91 """ 

92 try: 

93 # 獲取伺服器和頻道 

94 guild = self.bot.get_guild(guild_id) 

95 if not guild: 

96 return 

97 

98 channel = guild.get_channel(channel_id) 

99 if not channel or not isinstance(channel, discord.TextChannel): 

100 return 

101 

102 # 獲取排行榜資料 

103 rankings = await self.db.get_daily_rankings(ymd, guild_id, limit=5) 

104 if not rankings: 

105 return 

106 

107 # 獲取月度統計 

108 monthly_stats = await self.db.get_monthly_stats(ym, guild_id) 

109 

110 # 生成排行榜嵌入 

111 embed = self._create_ranking_embed(guild, rankings, monthly_stats, days) 

112 

113 # 發送排行榜 

114 await channel.send(embed=embed) 

115 

116 except Exception as e: 

117 logger.error(f"【活躍度】處理伺服器 {guild_id} 的排行榜播報失敗: {e}") 

118 

119 def _create_ranking_embed(self, guild: discord.Guild, rankings: List[Dict[str, Any]], 

120 monthly_stats: Dict[int, int], days: int) -> discord.Embed: 

121 """ 

122 創建排行榜嵌入 

123  

124 Args: 

125 guild: Discord 伺服器 

126 rankings: 排行榜資料 

127 monthly_stats: 月度統計資料 

128 days: 當月天數 

129  

130 Returns: 

131 discord.Embed: 排行榜嵌入 

132 """ 

133 lines = [] 

134 

135 for rank, data in enumerate(rankings, 1): 

136 user_id = data["user_id"] 

137 msg_cnt = data["msg_cnt"] 

138 

139 # 計算月平均 

140 mavg = monthly_stats.get(user_id, 0) / days if days else 0 

141 

142 # 獲取成員名稱 

143 member = guild.get_member(user_id) 

144 name = member.display_name if member else f"<@{user_id}>" 

145 

146 # 添加排行榜項目 

147 lines.append(f"`#{rank:2}` {name:<20} ‧ 今日 {msg_cnt} 則 ‧ 月均 {mavg:.1f}") 

148 

149 # 創建嵌入 

150 embed = discord.Embed( 

151 title=f"📈 今日活躍排行榜 - {guild.name}", 

152 description="\n".join(lines), 

153 colour=discord.Colour.green() 

154 ) 

155 

156 return embed
