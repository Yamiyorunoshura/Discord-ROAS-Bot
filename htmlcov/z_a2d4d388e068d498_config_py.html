119def get_file_extension(filename: str) -> str: 

120 """ 

121 獲取檔案副檔名 

122  

123 Args: 

124 filename: 檔案名稱 

125  

126 Returns: 

127 str: 副檔名（小寫，不包含點） 

128 """ 

129 try: 

130 if "." not in filename: 

131 return "" 

132 

133 # 處理多重副檔名（如 .tar.gz） 

134 parts = filename.lower().split(".") 

135 if len(parts) >= 3 and f"{parts[-2]}.{parts[-1]}" in STRICT_EXTENSIONS: 

136 return f"{parts[-2]}.{parts[-1]}" 

137 

138 return parts[-1] 

139 except Exception: 

140 return ""

142def is_dangerous_file(filename: str, strict_mode: bool = False) -> bool: 

143 """ 

144 檢查檔案是否為危險檔案 

145  

146 Args: 

147 filename: 檔案名稱 

148 strict_mode: 是否使用嚴格模式 

149  

150 Returns: 

151 bool: 是否為危險檔案 

152 """ 

153 try: 

154 ext = get_file_extension(filename) 

155 if not ext: 

156 return False 

157 

158 # 檢查基本危險副檔名 

159 if ext in DANGEROUS_EXTENSIONS: 

160 return True 

161 

162 # 嚴格模式額外檢查 

163 if strict_mode and ext in STRICT_EXTENSIONS: 

164 return True 

165 

166 return False 

167 except Exception: 

168 return False

201def extract_urls_from_text(text: str) -> List[str]: 

202 """ 

203 從文字中提取 URL 

204  

205 Args: 

206 text: 要檢查的文字 

207  

208 Returns: 

209 List[str]: 找到的 URL 列表 

210 """ 

211 try: 

212 # 簡單的 URL 檢測正則表達式 

213 url_pattern = re.compile(r'https?://[^\s<>"{}|\\^`\[\]]+', re.IGNORECASE) 

214 return url_pattern.findall(text) 

215 except Exception: 

216 return []

218def is_executable_url(url: str) -> bool: 

219 """ 

220 檢查 URL 是否指向可執行檔案 

221  

222 Args: 

223 url: 要檢查的 URL 

224  

225 Returns: 

226 bool: 是否指向可執行檔案 

227 """ 

228 try: 

229 # 從 URL 中提取檔案名稱 

230 import urllib.parse as up 

231 parsed = up.urlparse(url) 

232 path = parsed.path 

233 

234 if "/" in path: 

235 filename = path.split("/")[-1] 

236 else: 

237 filename = path 

238 

239 # 移除查詢參數 

240 if "?" in filename: 

241 filename = filename.split("?")[0] 

242 

243 return is_dangerous_file(filename) 

244 except Exception: 

245 return False

247def format_file_list(files: List[str], max_length: int = 1000) -> str: 

248 """ 

249 格式化檔案列表為字串 

250  

251 Args: 

252 files: 檔案列表 

253 max_length: 最大字串長度 

254  

255 Returns: 

256 str: 格式化後的字串 

257 """ 

258 try: 

259 if not files: 

260 return "（無）" 

261 

262 result = [] 

263 current_length = 0 

264 

265 for file in files: 

266 line = f"• {file}" 

267 if current_length + len(line) + 1 > max_length: 

268 remaining = len(files) - len(result) 

269 if remaining > 0: 

270 result.append(f"... 還有 {remaining} 個檔案") 

271 break 

272 

273 result.append(line) 

274 current_length += len(line) + 1 

275 

276 return "\n".join(result) 

277 except Exception: 

278 return "（解析錯誤）"

280def get_file_risk_level(filename: str, strict_mode: bool = False) -> str: 

281 """ 

282 獲取檔案風險等級 

283  

284 Args: 

285 filename: 檔案名稱 

286 strict_mode: 是否使用嚴格模式 

287  

288 Returns: 

289 str: 風險等級（高、中、低、安全） 

290 """ 

291 try: 

292 ext = get_file_extension(filename) 

293 if not ext: 

294 return "未知" 

295 

296 # 安全檔案 

297 if ext in SAFE_EXTENSIONS: 

298 return "安全" 

299 

300 # 高風險檔案 

301 high_risk = { 

302 "exe", "com", "scr", "bat", "cmd", "vbs", "js", "msi", "ps1" 

303 } 

304 if ext in high_risk: 

305 return "高" 

306 

307 # 中風險檔案 

308 if ext in DANGEROUS_EXTENSIONS: 

309 return "中" 

310 

311 # 嚴格模式下的可疑檔案 

312 if strict_mode and ext in STRICT_EXTENSIONS: 

313 return "低" 

314 

315 return "未知" 

316 except Exception: 

317 return "未知"
