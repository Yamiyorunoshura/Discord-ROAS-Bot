6import discord

9async def preview_embed(message_data: Dict[str, Any], bot: discord.Client) -> discord.Embed: 

10 """ 

11 生成訊息預覽的嵌入訊息 

12  

13 Args: 

14 message_data: 訊息資料 

15 bot: Discord 機器人實例 

16  

17 Returns: 

18 discord.Embed: 預覽嵌入訊息 

19 """ 

20 # 獲取基本資訊 

21 message_id = message_data.get("message_id") or message_data.get("id") 

22 channel_id = message_data.get("channel_id") 

23 author_id = message_data.get("author_id") 

24 content = message_data.get("content", "") 

25 timestamp = message_data.get("timestamp", 0) 

26 deleted = message_data.get("deleted", False) 

27 

28 # 獲取用戶和頻道 

29 user = bot.get_user(author_id) if author_id else None 

30 channel = bot.get_channel(channel_id) if channel_id else None 

31 

32 # 創建嵌入訊息 

33 embed = discord.Embed( 

34 title=f"{'🗑️ 已刪除的訊息' if deleted else '📝 訊息預覽'}", 

35 description=content or "[無內容]", 

36 color=discord.Color.red() if deleted else discord.Color.blue(), 

37 timestamp=discord.utils.utcnow() 

38 ) 

39 

40 # 添加作者資訊 

41 if user: 

42 embed.set_author( 

43 name=user.display_name, 

44 icon_url=user.display_avatar.url 

45 ) 

46 else: 

47 embed.set_author(name=f"用戶 {author_id}") 

48 

49 # 添加頻道資訊 

50 if channel: 

51 # 檢查頻道類型 

52 if isinstance(channel, discord.TextChannel): 

53 channel_value = f"{channel.mention} (`{channel.name}`)" 

54 else: 

55 # 處理其他頻道類型 

56 channel_name = getattr(channel, "name", "未知頻道") 

57 channel_value = f"`{channel_name}`" 

58 else: 

59 channel_value = f"未知頻道 ({channel_id})" 

60 

61 embed.add_field( 

62 name="頻道", 

63 value=channel_value, 

64 inline=True 

65 ) 

66 

67 # 添加訊息 ID 

68 if message_id: 

69 embed.add_field( 

70 name="訊息 ID", 

71 value=f"`{message_id}`", 

72 inline=True 

73 ) 

74 

75 # 添加時間資訊 

76 import datetime 

77 if timestamp: 

78 formatted_time = datetime.datetime.fromtimestamp(timestamp).strftime("%Y-%m-%d %H:%M:%S") 

79 embed.add_field( 

80 name="發送時間", 

81 value=formatted_time, 

82 inline=True 

83 ) 

84 

85 # 添加附件資訊 

86 attachments = message_data.get("attachments", []) 

87 if attachments: 

88 if isinstance(attachments, str): 

89 # 如果是 JSON 字串，嘗試解析 

90 import json 

91 try: 

92 attachments = json.loads(attachments) 

93 except: 

94 attachments = [] 

95 

96 if attachments: 

97 # 添加第一個附件的預覽 

98 first_attachment = attachments[0] 

99 if isinstance(first_attachment, dict): 

100 url = first_attachment.get("url") 

101 filename = first_attachment.get("filename") 

102 content_type = first_attachment.get("content_type") 

103 

104 if url and content_type and content_type.startswith("image/"): 

105 embed.set_image(url=url) 

106 

107 # 添加附件資訊 

108 attachment_info = [] 

109 for i, attachment in enumerate(attachments[:3]): 

110 if isinstance(attachment, dict): 

111 name = attachment.get("filename", "未知檔案") 

112 size = attachment.get("size", 0) 

113 size_str = f"{size / 1024:.1f} KB" if size else "未知大小" 

114 attachment_info.append(f"{i+1}. {name} ({size_str})") 

115 

116 if len(attachments) > 3: 

117 attachment_info.append(f"...以及 {len(attachments) - 3} 個附件") 

118 

119 if attachment_info: 

120 embed.add_field( 

121 name=f"附件 ({len(attachments)})", 

122 value="\n".join(attachment_info), 

123 inline=False 

124 ) 

125 

126 # 底部提示 

127 embed.set_footer(text=f"訊息 ID: {message_id}") 

128 

129 return embed
