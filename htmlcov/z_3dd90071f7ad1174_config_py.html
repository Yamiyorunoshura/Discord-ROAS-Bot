180def setup_logger(): 

181 """設定日誌記錄器""" 

182 logger = logging.getLogger("message_listener") 

183 if logger.handlers: 

184 return logger 

185 

186 logger.setLevel(logging.INFO) 

187 

188 # 檔案處理器 

189 file_handler = logging.FileHandler( 

190 MESSAGE_LOG_PATH, mode="a", encoding="utf-8" 

191 ) 

192 file_handler.setFormatter(logging.Formatter("%(asctime)s %(levelname)s %(message)s")) 

193 

194 # 控制台處理器 

195 console_handler = logging.StreamHandler() 

196 console_handler.setFormatter(logging.Formatter("%(asctime)s %(levelname)s %(message)s")) 

197 

198 logger.addHandler(file_handler) 

199 logger.addHandler(console_handler) 

200 

201 return logger

205 """檢查用戶是否有權限執行指定指令""" 

206 if not inter.guild: 

207 return False
