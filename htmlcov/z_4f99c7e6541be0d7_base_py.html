34def friendly_trace(exc: BaseException, depth: int = 3) -> str: 

35 """生成友善的錯誤追蹤資訊 

36  

37 Args: 

38 exc: 異常物件 

39 depth: 追蹤深度 

40  

41 Returns: 

42 格式化的錯誤追蹤字串 

43 """ 

44 return "".join(traceback.TracebackException.from_exception(exc, limit=depth).format())

47def friendly_log(msg: str, exc: Optional[BaseException] = None, level=logging.ERROR): 

48 """記錄友善的錯誤訊息，包含追蹤碼和詳細資訊 

49  

50 Args: 

51 msg: 錯誤訊息 

52 exc: 異常物件（可選） 

53 level: 日誌等級 

54 """ 

55 if exc: 

56 msg += f"\n原因：{exc.__class__.__name__}: {exc}\n{friendly_trace(exc)}" 

57 logger.log(level, msg, exc_info=bool(exc))

63@contextlib.contextmanager 

64def handle_error(ctx_or_itx: Optional[object], user_msg: str = "發生未知錯誤"): 

65 """統一的錯誤處理上下文管理器 

66  

67 Args: 

68 ctx_or_itx: Discord 上下文或互動物件 

69 user_msg: 顯示給使用者的錯誤訊息 

70 """ 

71 track = dt.datetime.utcnow().strftime("%Y%m%d%H%M%S%f") 

72 try: 

73 yield 

74 except Exception as exc: 

75 friendly_log(user_msg, exc) 

76 hint = f"❌ {user_msg}\n追蹤碼：`{track}`\n請將此碼提供給管理員。" 

77 try: 

78 if isinstance(ctx_or_itx, discord.Interaction): 

79 if ctx_or_itx.response.is_done(): 

80 asyncio.create_task(ctx_or_itx.followup.send(hint, ephemeral=True)) 

81 else: 

82 asyncio.create_task(ctx_or_itx.response.send_message(hint, ephemeral=True)) 

83 elif isinstance(ctx_or_itx, commands.Context): 

84 asyncio.create_task(ctx_or_itx.reply(hint, mention_author=False)) 

85 except Exception: 

86 pass

92class ProtectionCog(commands.Cog): 

93 """ 

94 群組保護模組的基礎類別 

95  

96 提供以下功能： 

97 - 資料庫配置管理（get_cfg / set_cfg） 

98 - 日誌記錄（log） 

99 - 權限檢查 

100 - 快取機制 

101  

102 子類別需要覆寫 module_name 屬性 

103 """ 

104 module_name: str = "base" 

105 

106 def __init__(self, bot: commands.Bot): 

107 """初始化保護模組 

108  

109 Args: 

110 bot: Discord Bot 實例 

111 """ 

112 self.bot = bot 

113 self._cache: dict[int, dict[str, str]] = {} # 配置快取 

114 

115 # ───────── 資料庫操作 ───────── 

116 async def _ensure_table(self): 

117 """確保保護配置資料表存在""" 

118 sql = """ 

119 CREATE TABLE IF NOT EXISTS protection_config( 

120 guild_id INTEGER, 

121 module TEXT, 

122 key TEXT, 

123 value TEXT, 

124 PRIMARY KEY (guild_id, module, key) 

125 );""" 

126 database = getattr(self.bot, 'database', None) 

127 if database: 

128 await database.execute(sql) 

129 

130 async def get_cfg(self, gid: int, key: str, default: Optional[str] = None) -> Optional[str]: 

131 """取得伺服器配置值 

132  

133 Args: 

134 gid: 伺服器 ID 

135 key: 配置鍵名 

136 default: 預設值 

137  

138 Returns: 

139 配置值或預設值 

140 """ 

141 if gid not in self._cache: 

142 await self._ensure_table() 

143 database = getattr(self.bot, 'database', None) 

144 if database: 

145 rows = await database.fetchall( 

146 "SELECT key,value FROM protection_config WHERE guild_id=? AND module=?", 

147 (gid, self.module_name), 

148 ) 

149 self._cache[gid] = {r["key"]: r["value"] for r in rows} 

150 else: 

151 self._cache[gid] = {} 

152 return self._cache[gid].get(key, default) 

153 

154 async def set_cfg(self, gid: int, key: str, value: str): 

155 """設定伺服器配置值 

156  

157 Args: 

158 gid: 伺服器 ID 

159 key: 配置鍵名 

160 value: 配置值 

161 """ 

162 await self._ensure_table() 

163 database = getattr(self.bot, 'database', None) 

164 if database: 

165 await database.execute( 

166 "INSERT OR REPLACE INTO protection_config VALUES (?,?,?,?)", 

167 (gid, self.module_name, key, value), 

168 ) 

169 self._cache.setdefault(gid, {})[key] = value

171 # ───────── 日誌記錄 ───────── 

172 async def log(self, guild: discord.Guild, msg: str): 

173 """記錄保護事件到指定頻道 

174  

175 Args: 

176 guild: Discord 伺服器 

177 msg: 日誌訊息 

178 """ 

179 try: 

180 chan_id = await self.get_cfg(guild.id, "log_channel") 

181 if chan_id and chan_id.isdigit(): 

182 ch = guild.get_channel(int(chan_id)) 

183 if ch and isinstance(ch, discord.TextChannel): # 確保是文字頻道 

184 await ch.send(f"[{self.module_name}] {msg}") 

185 except Exception as exc: 

186 friendly_log(f"記錄日誌失敗（伺服器：{guild.id}）", exc)

188 # ───────── 權限檢查 ───────── 

189 @staticmethod 

190 def is_admin(): 

191 """檢查是否為伺服器管理員（用於 prefix 指令）""" 

192 def pred(ctx: commands.Context) -> bool: 

193 if not ctx.guild: 

194 return False 

195 # 透過 guild 獲取成員物件來檢查權限 

196 member = ctx.guild.get_member(ctx.author.id) 

197 if member is None: 

198 return False 

199 return member.guild_permissions.manage_guild 

200 return commands.check(pred)
